SELECT * FROM ISSUES;

SELECT * FROM ISSUELISTS;

SELECT I.SUMMARY, I.DESCRIPTION, IL.NAME FROM ISSUES I
JOIN ISSUELISTS IL ON I.ISSUELIST_ID = IL.ID;

SELECT * FROM USERS;

SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME FROM ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;

SELECT * FROM ISSUES;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(I.SUMMARY) AS ISSUES_NUMBER
FROM ISSUES I JOIN USERS U on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY USER_ID_ASSIGNEDTO;
